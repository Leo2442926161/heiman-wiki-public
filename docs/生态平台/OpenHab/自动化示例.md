# 自动化示例

## 烟雾报警规则

```javascript
rule "烟雾报警触发"
when
    Item heiman_smoke_alarm changed to ON
then
    sendCommand(Livingroom_Light, ON)
    sendNotification("admin", "⚠️ 烟雾报警器触发！")
    postUpdate(Alarm_State, "smoke_alert")
end
```

## 温度传感器规则

```javascript
rule "温湿度上报"
when
    Item heiman_temperature received update
then
    val temp = (heiman_temperature.state as Number).floatValue
    val hum = (heiman_humidity.state as Number).floatValue
    if (temp > 40) {
        sendCommand(AirConditioner, ON)
        sendNotification("admin", "温度过高(" + temp + "°C)，已开启空调")
    }
    if (hum > 80) {
        sendNotification("admin", "湿度过高(" + hum + "%)，请注意除湿")
    }
end
```

## 水浸检测规则

```javascript
rule "水浸检测联动"
when
    Item heiman_water_leak changed to ON
then
    sendCommand(Water_Valve, OFF)
    sendBroadcastNotification("💧 检测到漏水！已关闭水阀！")
end
```
